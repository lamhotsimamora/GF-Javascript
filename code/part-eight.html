For set arrow of keyboard event ( Up, Down, Left & Right )
_keyArrow(function(){ _printTo("display_key","LEFT"); },function(){ _printTo("display_key","RIGHT"); },function(){ _printTo("display_key","UP"); },function(){ _printTo("display_key","DOWN"); });

For set custom keyboard event
_keyCustom(function(){ _printTo("display_key","BACKSPACE"); },_keyCode.backspace);
